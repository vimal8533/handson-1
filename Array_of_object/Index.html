Q6. This time we are required to print the sum of marks of the students with id > 120.

    Ans:
    function markssum(record){
        let sum=0
        for(let i=0 ; i< record.length;i++ ){
            if(record[i].id>120){
                sum+=record[i].marks;
            }
        }
        return sum
        }
        console.log(markssum(studentRecords));
    Output:143

Q7. This time we are required to print the total marks of the students with marks greater than 50 after a grace of 15 marks has been added to those students who scored less than 50.

Ans:
    function markssum50(record){
        let sum1=0
        for(let i=0 ; i< record.length;i++){
            if(record[i].marks>50){
                sum1+=record[i].marks;
            }else if(record[i].marks< 50){
                sum1+=record[i].marks+15
            }
        }
        return sum1
        }
    console.log(markssum50(studentRecords));

    Output:271
